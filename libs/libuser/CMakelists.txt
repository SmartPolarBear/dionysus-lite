cmake_minimum_required(VERSION 3.16 FATAL_ERROR)

add_library(libuser STATIC uentry.S syscall_client.S uentry_main.cc)

set_property(SOURCE uentry.S PROPERTY LANGUAGE C)
set_property(SOURCE syscall_client.S PROPERTY LANGUAGE C)

target_sources(libuser PRIVATE uentry.S syscall_client.S uentry_main.cc)

add_subdirectory(kernel_api)

target_include_directories(libuser PRIVATE "${CMAKE_SOURCE_DIR}/include")
